33/**
34 * Auto-inject Schema JSON-LD into page head
35 * This is the core functionality - automatically adds schema markup
36 */
37function svp_inject_schema() {
38    if (!is_singular()) {
39        return; // Only inject on single posts/pages
40    }
41
42    global $post;
43    if (!$post) {
44        return;
45    }
46
47    // Get stored schema from post meta
48    $schema = get_post_meta($post->ID, '_svp_schema', true);
49
50    if (empty($schema)) {
51        return;
52    }
53
54    // Decode if it's JSON string
55    if (is_string($schema)) {
56        $schema_data = json_decode($schema, true);
57    } else {
58        $schema_data = $schema;
59    }
60
61    if (empty($schema_data)) {
62        return;
63    }
64
65    // Check if schema already exists in the page (prevent duplicate injection)
66    if (svp_has_existing_schema()) {
67        return;
68    }
69
70    // Allow filtering schema data before injection
71    $schema_data = apply_filters('svp_schema_data', $schema_data, $post->ID);
72
73    // Fire action before injection
74    do_action('svp_before_schema_injection', $schema_data, $post->ID);
75
76    // Output schema as JSON-LD (using wp_json_encode for security)
77    echo "\n<!-- Schema Validator Pro v" . esc_attr(SCHEMA_VALIDATOR_PRO_VERSION) . " -->\n";
78    echo '<script type="application/ld+json">';
79    echo wp_json_encode($schema_data, JSON_UNESCAPED_SLASHES | JSON_UNESCAPED_UNICODE | JSON_PRETTY_PRINT);
80    echo '</script>';
81    echo "\n<!-- /Schema Validator Pro -->\n";
82
83    // Fire action after injection
84    do_action('svp_after_schema_injection', $schema_data, $post->ID);
85}
